# Firebase-konfigurasjon
# Kopier denne filen til .env og fyll inn dine Firebase-verdier
# Du finner disse verdiene i Firebase Console under Project Settings > Your apps > Web app

FIREBASE_API_KEY=din-api-key-her
FIREBASE_AUTH_DOMAIN=ditt-prosjekt.firebaseapp.com
FIREBASE_PROJECT_ID=ditt-prosjekt-id
FIREBASE_STORAGE_BUCKET=ditt-prosjekt.firebasestorage.app
FIREBASE_MESSAGING_SENDER_ID=123456789
FIREBASE_APP_ID=1:123456789:web:abcdef123456

# VIKTIG: 
# - Aldri commit .env-filen til Git (den er allerede i .gitignore)
# - For GitHub Pages deploy, sett opp disse verdiene som Secrets i GitHub
# - Se FIREBASE_SETUP.md for detaljerte instruksjoner
